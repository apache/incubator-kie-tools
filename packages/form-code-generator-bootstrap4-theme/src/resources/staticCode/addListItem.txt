function addListItem(value, defaultValue, disabled, maxCount, name, childrenHtml) {
    return () => {
      if (!disabled && maxCount > value.length) {
        value = value.concat([defaultValue]);
      }

      const itemIndex = value.length-1;

      const listContainer = document.getElementById("list-" + name);

      const newItem = document.createElement("div");
      newItem.class = "row"
      newItem.id = name + "-" + itemIndex
      newItem.innerHTML = `
      <div class="col-1">
        <span class="badge badge-pill" role="button" tabindex="0" onclick="delListItem(itemIndex)" onkeydown="delListItem(itemIndex)">
          <i className="octicon octicon-dash" />
        </span>
      </div>
      <div class="col-11">
        ${childrenHtml}
      </div>
      `
      }
    
    listContainer.appendChild(newItem);
}
