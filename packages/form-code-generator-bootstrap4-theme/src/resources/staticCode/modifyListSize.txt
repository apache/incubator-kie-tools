function delListItem(value, disabled, minCount, itemIndex) {
  if (!disabled && minCount < value.length) {
    const newValue = [...value];
    newValue.splice(itemIndex, 1);
    value = newValue;
  }
}

function addListItem(value, defaultValue, disabled, maxCount, name, itemIndex, childrenHtml) {
  if (!disabled && maxCount > value.length) {
    value = value.concat([defaultValue]);
  }

  const listContainer = document.getElementById(name);

  const newItem = document.createElement("div");
  newItem.class = "row"
  newItem.id = name + "." + itemIndex
  newItem.innerHTML = `
  <div class="col-1">
    <span class="badge badge-pill" role="button" tabindex="0" onclick="onDelListItem('${name}', ${itemIndex})" onkeydown="onDelListItem('${name}', ${itemIndex})">
      <i className="octicon octicon-dash"><span>-</span></i>
    </span>
  </div>
  <div class="col-11">
    ${childrenHtml}
  </div>
  `

  listContainer.appendChild(newItem);
}
